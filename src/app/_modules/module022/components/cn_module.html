022 Creating a custom attribute Directive

HTML
Whats going on?
  • used as attribute
    <input credit-card>
  • bare definition
    import { Directive, ElementRef } from '@angular/core';

    @Directive({
      selector: '[credit-card]'
    })
    export class CreditCardDirective {
      constructor(private element: ElementRef) {
        console.log(element);
      } }
  • not to forget to include into the module declaration
    import { CreditCardDirective } from './components/di_credit-card.directive';

    @NgModule({
      declarations: [
        ...,
        CreditCardDirective,
      ],
      imports: [ ... ],
    })
    export class ModuleModule {}
